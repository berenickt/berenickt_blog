## 1. Link 컴포넌트

- cf. https://nextjs.org/docs/app/api-reference/components/link

```tsx
<Link
  // 해당 컴포넌트가 연결될 주소 (required)
  href="/home"
  // 링크로 이동시 history.replace()를 사용할지 여부 (default: false)
  replace={false}
  // 링크 이동 후 페이지 스크롤 초기화 여부 (default: true)
  scroll={true}
  // 링크 컴포넌트가 Viewport에 표시될 경우 페이지 Prefetch 여부 (default: true)
  prefetch={true}
/>
```
